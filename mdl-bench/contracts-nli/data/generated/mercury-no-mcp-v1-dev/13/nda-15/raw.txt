@ Document: NON‑DISCLOSURE AGREEMENT FOR THE EXCHANGE OF ENERGY MANAGEMENT SYSTEM MODEL DATA
@ Hypothesis: “Agreement shall not grant Receiving Party any right to Confidential Information.”
module contract_nli_no_align_dev_13_nda_15

# ----------------------------------------------------------------------
# Types
# ----------------------------------------------------------------------
@ Parties to the Agreement (PJM and a Transmission Owner)
type Party = {
    name: string,               # e.g. "PJM Interconnection" or "Transmission Owner"
    is_disclosing: bool         # true for the Disclosing Party, false for the Receiving Party
}

    data: string                # opaque representation of the model data
}

@ An employee or agent who may view the confidential data
type ReviewingRepresentative = {
    name: string,

    has_certificate: bool,      # true when a signed Non‑Disclosure Certificate has been submitted
    duties: List<string>        # e.g. ["TransmissionFunctions", "CompetitiveDuties"]
}

    signed: bool,
    date_signed: string
}

@ Receiving Party (Transmission Owner)

func has_competitive_duty(rep: ReviewingRepresentative) -> bool:
    list.contains(rep.duties, "CompetitiveDuties")

    false   # default to false; can be overridden by a fact in a concrete instance

    (has_competitive_duty(reviewing_rep) or not reviewing_rep.has_certificate) implies
        (confidential_information.owner = disclosing_party) and
        (not reviewing_rep.party = receiving_party) always

@ 2.7 – The Receiving Party must return or destroy Confidential Information

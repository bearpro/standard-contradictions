module contract_nli_no_align_dev_13_nda_5

@ Clause 1 – Parties
type Party =

@ Clause 5.4 – Confidential Information
type ConfidentialInfo = {
    owner: Party,               # Party that supplied the information
    marked_confidential: bool   # true when the material is marked “Confidential”
}

    is_employee_of_receiving_party: bool,   # true for any employee of the Receiving Party

    person.is_reviewing_rep implies person.has_executed_certificate always

    (person.is_reviewing_rep and person.has_executed_certificate) always

    (person.is_employee_of_receiving_party and not person.is_reviewing_rep) always

    confidential_info.marked_confidential always

@ Determines whether a particular disclosure is allowed under any

func allowed_exceptions(info: ConfidentialInfo, role: ThirdPartyRole) -> bool:
    let consent      = info.is_consent_given
    let performance  = info.is_needed_for_performance
    let public       = info.is_publicly_available
    let already      = info.is_already_possessed_by_receiving
    let independent  = info.is_developed_independently
    let third_party  = info.is_rightfully_received_from_third_party

        | Agent():              consent or performance or public or already or independent or third_party
        | ProfessionalAdvisor(): consent or performance or public or already or independent or third_party

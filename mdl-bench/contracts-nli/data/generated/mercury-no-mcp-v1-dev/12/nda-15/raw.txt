func is_allowed_disclosure(r: DisclosureReason) -> bool:

        | Consent(_):               true
        | LegalCompulsion(_):       true
        | ServicePerformance(_):    true
        | PublicAvailability(_):    true
        | IndependentDevelopment(_): true
        | AlreadyPossessed(_):      true
        | ThirdPartyWithoutObligation(_): true

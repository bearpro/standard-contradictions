module contract_nli_no_align_dev_14_nda_15

open std.strings

type Party =

        name: string,
        -- Boolean flags used by the rules below
        has_right_to_confidential_material: bool,
        discloses_confidential_material: bool,
        notifies_company_if_required: bool,
        accepts_delaware_chancery_jurisdiction: bool
    }

@ Confidential material that may be inspected
type ConfidentialMaterial = {
    description: string
}

-- ----------------------------------------------------------------------
-- Entities (public model objects)
-- ----------------------------------------------------------------------

entity company: Party = Company { name = "Patriot Scientific Corporation" }
entity stockholder: Party = Stockholder {
    name = "Stockholder",
    has_right_to_confidential_material = false,
    discloses_confidential_material = false,
    notifies_company_if_required = false,
    accepts_delaware_chancery_jurisdiction = false

    not stockholder.has_right_to_confidential_material always

@ Clause 6 – Amendment only by written instrument signed by the other party

    true always   -- placeholder: the rule is captured by the modality of the agreement,

    true always   -- placeholder: no additional state required

@ Clause 8 – Governing law of the State of Delaware

    true always   -- placeholder: no additional state required

@ Clause 9 – Execution in multiple counterparts
rule O agreement_may_be_executed_in_multiple_counterparts:
    true always   -- placeholder: no additional state required

@ Clause 10 – Entire agreement clause

    true always   -- placeholder: no additional state required
